<!-- wp:paragraph -->

Exits Consteel.

<!-- /wp:paragraph -->

<!-- wp:heading {"level":3} -->

### Syntax

<!-- /wp:heading -->

<!-- wp:paragraph -->

**EXIT**

<!-- /wp:paragraph -->

<!-- wp:heading {"level":3} -->

### Description

<!-- /wp:heading -->

<!-- wp:paragraph -->

Closes the Consteel window. If there are any unsaved changes is the model or in the script editor, a dialogue window will pop up asking if it should save before closing the program.

<!-- /wp:paragraph -->

<!-- wp:heading {"level":3} -->

### Sample code:

<!-- /wp:heading -->

<!-- wp:loos-hcb/code-block -->

```
EXIT
```

<!-- /wp:loos-hcb/code-block -->
